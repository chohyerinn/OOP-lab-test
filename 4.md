## Problem 1. RoomOccupancy

```java
public class RoomOccupancy {
    private int roomNumber;
    private int peopleInRoom;
    private static int totalNumber = 0;

    public RoomOccupancy(int roomNumber, int peopleInRoom) {
        this.roomNumber = roomNumber;
        this.peopleInRoom = peopleInRoom;
        totalNumber += peopleInRoom;
    }

    public void addOneToRoom() {
        peopleInRoom++;
        totalNumber++;
    }

    public void removeOneFromRoom() {
        if (peopleInRoom > 0) {
            peopleInRoom--;
            totalNumber--;
        }
    }

    public int getNumber() {
        return roomNumber;
    }

    public int getPeopleInRoom() {
        return peopleInRoom;
    }

    public static int getTotal() {
        return totalNumber;
    }
}
```

## Problem 3. MotorBoat

```java
public class MotorBoat {
    public static final double TANK_CAPACITY = 60.0;
    public static final double MAX_SPEED = 80.0;

    private double fuel;
    private double currentSpeed;
    private double efficiency;

    public MotorBoat() {
        fuel = 0;
        currentSpeed = 0;
        efficiency = 1;
    }

    public MotorBoat(double fuel, double currentSpeed, double efficiency) {
        setFuel(fuel);
        setCurrentSpeed(currentSpeed);
        setEfficiency(efficiency);
    }

    public void setFuel(double fuel) {
        if (fuel >= 0 && fuel <= TANK_CAPACITY) this.fuel = fuel;
    }

    public void setCurrentSpeed(double currentSpeed) {
        if (currentSpeed >= 0 && currentSpeed <= MAX_SPEED) this.currentSpeed = currentSpeed;
    }

    public void setEfficiency(double efficiency) {
        if (efficiency > 0) this.efficiency = efficiency;
    }

    public double fuelUsed(double time) {
        return efficiency * currentSpeed * currentSpeed * time;
    }

    public double travelDistance(double time) {
        return currentSpeed * time;
    }

    public double travelDistanceWithCurrentFuel() {
        double time = fuel / (efficiency * currentSpeed * currentSpeed);
        return travelDistance(time);
    }
}
```

```java
import java.util.Scanner;

public class SalesReporter {
    private double highestSales;
    private double averageSales;
    private SalesAssociate[] team;
    private int numberOfAssociates;

    public void getData() {
        Scanner keyboard = new Scanner(System.in);
        System.out.print("Enter number of associates: ");
        numberOfAssociates = keyboard.nextInt();
        team = new SalesAssociate[numberOfAssociates];

        for (int i = 0; i < team.length; i++) {
            team[i] = new SalesAssociate();
            team[i].readInput();
        }
    }

    public void computeStats() {
        double sum = 0;
        highestSales = team[0].getSales();

        for (int i = 0; i < team.length; i++) {
            double sales = team[i].getSales();
            sum += sales;
            if (sales > highestSales) highestSales = sales;
        }

        averageSales = sum / team.length;
    }

    public void displayResults() {
        System.out.println("Average sales = " + averageSales);
        System.out.println("Highest sales = " + highestSales);

        for (int i = 0; i < team.length; i++) {
            if (team[i].getSales() == highestSales) {
                team[i].writeOutput();
            }
        }
    }
}
```

## Problem 6. TimeBook

```java
public class TimeBook {
    private int numberOfEmployees;
    private int[][] hours;
    private int[] weekHours;
    private int[] dayHours;
    private static final int NUMBER_OF_WORKDAYS = 5;

    public TimeBook(int numberOfEmployees) {
        this.numberOfEmployees = numberOfEmployees;
        hours = new int[NUMBER_OF_WORKDAYS][numberOfEmployees];
        weekHours = new int[numberOfEmployees];
        dayHours = new int[NUMBER_OF_WORKDAYS];
    }

    public void update() {
        computeWeekHours();
        computeDayHours();
    }

    private void computeWeekHours() {
        for (int emp = 0; emp < numberOfEmployees; emp++) {
            int sum = 0;
            for (int day = 0; day < NUMBER_OF_WORKDAYS; day++) {
                sum += hours[day][emp];
            }
            weekHours[emp] = sum;
        }
    }

    private void computeDayHours() {
        for (int day = 0; day < NUMBER_OF_WORKDAYS; day++) {
            int sum = 0;
            for (int emp = 0; emp < numberOfEmployees; emp++) {
                sum += hours[day][emp];
            }
            dayHours[day] = sum;
        }
    }
}
```

## Problem 7. FigureUtility

```java
public class FigureUtility {
    public static double[][] findFigure(double[][] picture, double threshold) {
        double avg = average(picture);
        double[][] result = new double[picture.length][picture[0].length];

        for (int r = 0; r < picture.length; r++) {
            for (int c = 0; c < picture[r].length; c++) {
                if (picture[r][c] > avg * threshold) {
                    result[r][c] = 1.0;
                } else {
                    result[r][c] = 0.0;
                }
            }
        }

        return result;
    }

    private static double average(double[][] picture) {
        double sum = 0;
        int count = 0;

        for (int r = 0; r < picture.length; r++) {
            for (int c = 0; c < picture[r].length; c++) {
                sum += picture[r][c];
                count++;
            }
        }

        return sum / count;
    }
}
```

## Problem 8. FlowerOrderUtility

```java
public class FlowerOrderUtility {
    public static boolean isStrictlyDecreasing(int[] counts) {
        for (int i = 1; i < counts.length; i++) {
            if (counts[i] >= counts[i - 1]) return false;
        }
        return true;
    }

    public static void replaceAll(String[] order, String oldFlower, String newFlower) {
        for (int i = 0; i < order.length; i++) {
            if (order[i].equals(oldFlower)) {
                order[i] = newFlower;
            }
        }
    }

    public static String[] combineOrders(String[] first, String[] second) {
        String[] result = new String[first.length + second.length];

        for (int i = 0; i < first.length; i++) result[i] = first[i];
        for (int i = 0; i < second.length; i++) result[first.length + i] = second[i];

        return result;
    }
}
```

```java
public class Doctor extends Person {

        super();

        visitFee = 0.0;

        super(name);
        setSpecialty(specialty);
        setVisitFee(visitFee);
    }

    public void setSpecialty(String specialty) {
        if (specialty.equals("Medicine") || specialty.equals("Surgery")
                || specialty.equals("Dentist") || specialty.equals("Oriental")) {
            this.specialty = specialty;
        } else {
            this.specialty = "Medicine";
        }
    }

    public void setVisitFee(double visitFee) {
        if (visitFee >= 0) this.visitFee = visitFee;
        else this.visitFee = 0.0;
    }

        return "Specialty: " + specialty + "\nVisit fee: " + visitFee;
    }

    public boolean equals(Object otherObject) {
        if (!(otherObject instanceof Doctor)) return false;
        Doctor other = (Doctor) otherObject;
        return specialty.equals(other.specialty) && visitFee == other.visitFee;
    }
```

```java
            throws DivideByZeroException, UnknownOpException {
        if (op == '+') return n1 + n2;
        if (op == '-') return n1 - n2;
        if (op == '*') return n1 * n2;
```

```java
import java.io.BufferedReader;
import java.io.FileReader;
import java.io.FileWriter;
import java.io.IOException;
import java.io.PrintWriter;

public class LineCopier {
    private String sourceFile;
    private String destinationFile;

    public LineCopier(String sourceFile, String destinationFile) {
        this.sourceFile = sourceFile;
        this.destinationFile = destinationFile;
    }

    public int copyLines() throws IOException {
        BufferedReader input = new BufferedReader(new FileReader(sourceFile));
        PrintWriter output = new PrintWriter(new FileWriter(destinationFile));
        int count = 0;
        String line;

        while ((line = input.readLine()) != null) {
            output.println(line);
            count++;
        }

        input.close();
        output.close();
        return count;
    }
}
```

## Problem 16. CSV SalesFileAnalyzer

```java
import java.io.File;
import java.io.FileNotFoundException;
import java.util.Scanner;

public class SalesFileAnalyzer {
    private String fileName;
    private double totalSales;
    private int recordCount;

    public SalesFileAnalyzer(String fileName) {
        this.fileName = fileName;
    }

    public void processFile() throws FileNotFoundException {
        Scanner input = new Scanner(new File(fileName));

        while (input.hasNextLine()) {
            String line = input.nextLine();
            String[] parts = line.split(",");
            int quantity = Integer.parseInt(parts[1]);
            double price = Double.parseDouble(parts[2]);
            totalSales += quantity * price;
            recordCount++;
        }

        input.close();
    }

    public void printSummary() {
        System.out.println("Records: " + recordCount);
        System.out.println("Total sales: " + totalSales);
    }
}
```

## Problem 17. SavingsAccount

```java
public class SavingsAccount {
    private double balance;
    public static final double INTEREST_RATE = 2.9;
    private static int numberOfAccounts = 0;

    public SavingsAccount() {
        this(0);
    }

    public SavingsAccount(double balance) {
        setBalance(balance);
        numberOfAccounts++;
    }

    public void setBalance(double balance) {
        if (balance >= 0) this.balance = balance;
        else this.balance = 0;
    }

    public void deposit(double amount) {
        if (amount > 0) balance += amount;
    }

    public double withdraw(double amount) {
        if (amount <= 0) return 0;
        if (amount > balance) {
            double temp = balance;
            balance = 0;
            return temp;
        }
        balance -= amount;
        return amount;
    }

    public void addInterest() {
        balance += balance * INTEREST_RATE / 100;
    }

    public static int getNumberOfAccounts() {
        return numberOfAccounts;
    }
}
```

```java
public class AverageUtility {

    }
```
